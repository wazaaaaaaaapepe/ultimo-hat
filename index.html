const express = require('express'); const axios = require('axios'); const app = express(); app.get('/limiteds', async (req, res) => { try { const response = await axios.get('https://www.rolimons.com/itemapi/itemdetails'); const data = response.data; const limitedItems = data.filter(item => item.isLimited); if (limitedItems.length > 0) { const item = limitedItems[0]; res.send(`

Nombre: ${item.name}

Precio: ${item.price}

Link: Ver en Roblox

`); } else { res.send("No se encontraron Limiteds."); } } catch (error) { res.status(500).send("Error al obtener datos: " + error.message); } }); app.listen(3000, () => console.log('Servidor corriendo en http://localhost:3000'));
