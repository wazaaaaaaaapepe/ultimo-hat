const noblox = require('noblox.js'); noblox.setCookie('tu_cookie_de_roblox').then(async () => { const items = await noblox.getCatalogItems({ limit: 30 }); const limiteds = items.filter(item => item.collectibleItemId); if (limiteds.length > 0) { const item = limiteds[0]; console.log(`Nombre: ${item.name}`); console.log(`Precio: ${item.price}`); console.log(`Link: https://www.roblox.com/catalog/${item.id}`); } else { console.log("No se encontraron Limiteds."); } });
