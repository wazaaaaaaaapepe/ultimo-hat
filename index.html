const express = require('express'); const axios = require('axios'); const app = express(); app.get('/limiteds', async (req, res) => { try { const response = await axios.get('https://cors-anywhere.herokuapp.com/https://catalog.roblox.com/v1/search/items/details?Category=11&SortType=3&Limit=30'); const data = response.data; const limitedItems = data.data.filter(item => item.collectibleItemId); if (limitedItems.length > 0) { const item = limitedItems[0]; res.send(`

Nombre: ${item.name}

Precio: ${item.price || "Gratis"}

Link: Ver en Roblox

`); } else { res.send("No se encontraron Limiteds."); } } catch (error) { res.status(500).send("Error al obtener datos: " + error.message); } }); app.listen(3000, () => console.log('Servidor corriendo en http://localhost:3000'));
